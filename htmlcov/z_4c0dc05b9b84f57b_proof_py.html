236 @classmethod 

237 def from_rule( 

238 cls, 

239 rule: "Rule", 

240 *, 

241 conclusions: list[ProofPayload], 

242 premises: list[ProofNode] | None = None, 

243 derivation: DerivationRecord | None = None, 

244 rationale: str | None = None, 

245 stated_theorem_name: str | None = None, 

246 ) -> "RuleApplication": 

247 """Construct a proof step from a defined rule and its shared description.""" 

248 return cls( 

249 conclusions=conclusions, 

250 premises=[] if premises is None else premises, 

251 rule_id=rule.id, 

252 description=rule.description, 

253 derivation=derivation, 

254 rationale=rationale, 

255 stated_theorem_name=stated_theorem_name, 

256 )

279 @model_validator(mode="after") 

280 def check_derivation_alignment(self) -> "RuleApplication": 

281 if self.rule_id is None and self.description is None: 

282 raise ValueError( 

283 "RuleApplication MUST provide at least one of `rule_id` or `description`." 

284 ) 

285 if ( 

286 self.derivation is not None 

287 and self.rule_id is not None 

288 and self.rule_id != self.derivation.rule_id 

289 ): 

290 raise ValueError( 

291 "RuleApplication.rule_id MUST match DerivationRecord.rule_id " 

292 "when a derivation record is attached and a rule_id is provided." 

293 ) 

294 return self
